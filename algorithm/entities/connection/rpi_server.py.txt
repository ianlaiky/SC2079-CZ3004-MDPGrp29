import socket
import pickle


class RPiServer:
    """
    Used as the server in the RPi.
    """
    def __init__(self, host, port):
        self.host = host
        self.port = port
        self.socket = socket.socket()

        self.__data = []
        self.conn, self.address = None, None

    def start(self):
        print(f"Creating server at {self.host}:{self.port}")
        self.socket.bind((self.host, self.port))
        self.socket.listen()
        print("Listening for connection...")

        self.conn, self.address = self.socket.accept()
        print(f"1 Connection from {self.address}")

    def receive_data(self):
        assert self.conn is not None and self.address is not None
        with self.conn:
            print(f"2 Connection from {self.address}")

            d = self.conn.recv(65536)
            return(pickle.loads(d))

    def close(self):
        print("Closing socket.")
        self.socket.close()
